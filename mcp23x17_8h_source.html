15enum class MCPRegister : uint8_t

210 void PinMode(uint8_t pin, MCPMode mode, bool inverted)
211 {
212 MCPRegister iodirreg = MCPRegister::IODIR_A;
213 MCPRegister pullupreg = MCPRegister::GPPU_A;
214 MCPRegister polreg = MCPRegister::IPOL_A;
215 uint8_t iodir, pol, pull;
216 if(pin > 7)
217 {
218 iodirreg = MCPRegister::IODIR_B;
219 pullupreg = MCPRegister::GPPU_B;
220 polreg = MCPRegister::IPOL_B;
221 pin -= 8;
222 }
223 iodir = transport.ReadReg(iodirreg);
224 if(mode == MCPMode::INPUT || mode == MCPMode::INPUT_PULLUP)
225 SetBit(iodir, pin);
226 else
227 ClearBit(iodir, pin);
228 pull = transport.ReadReg(pullupreg);
229 if(mode == MCPMode::INPUT_PULLUP)
230 SetBit(pull, pin);
231 else
232 ClearBit(pull, pin);
233 pol = transport.ReadReg(polreg);
234 if(inverted)
235 SetBit(pol, pin);
236 else
237 ClearBit(pol, pin);
238 transport.WriteReg(iodirreg, iodir);
239 transport.WriteReg(pullupreg, pull);
240 transport.WriteReg(polreg, pol);
241 }
