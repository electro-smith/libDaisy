79 BlockingSpiTransport::Result
80 Transmit(uint8_t* buff, size_t size, uint32_t wait_us)
81 {
82 if(wait_us > 0)
83 {
84 uint32_t ts = System::GetUs();
85 // Since this is a transaction which requires a delay
86 // we check if enough time has elapsed, if not, we simply
87 // return "OK".
88 if(ts < next_tx_)
89 {
90 // Check if the clock rolled over, the max wait time is 20*40us
91 if((next_tx_ - ts) > (20 * 40))
92 {
93 next_tx_ = ts + wait_us;
94 return Result::OK;
95 }
96 else
97 {
98 return Result::OK;
99 }
100 }
101 }
102
103 if(spi_.BlockingTransmit(buff, size) == SpiHandle::Result::ERR)
104 {
105 return Result::ERR;
106 }
107
108 if(wait_us > 0)
109 {
110 // Reset next_tx timestamp
111 next_tx_ = System::GetUs() + wait_us;
112 }
113 return Result::OK;
114 }
115
116 BlockingSpiTransport::Result
117 TransmitAndReceive(uint8_t* tx_buff, uint8_t* rx_buff, size_t size)
118 {
119 if(spi_.BlockingTransmitAndReceive(tx_buff, rx_buff, size)
120 == SpiHandle::Result::ERR)
121 {
122 return Result::ERR;
123 }
124
125 return Result::OK;
126 }

243 Result Init(Config config)
244 {
245 transport_.Init(config.transport_config);
246
247 // First, let's verify the SPI comms, and chip presence. The DEVID register
248 // is a fixed, read-only value we can compare against to ensure we're connected.
249 if(ReadRegister(MAX11300_DEVICE_ID) != 0x0424)
250 {
251 return Result::ERR;
252 }
253
254 // Init routine (roughly) as per the datasheet pp. 49
255 // These settings were chosen as best applicable for use in a Eurorack context.
256 // Should the need for more configurability arise, this would be the spot to do it.
257
258 // Setup the device...
259 uint16_t devctl = 0x0000;
260 // 1:0 ADCCTL[1:0] - ADC conversion mode selection = 11: Continuous sweep
261 devctl = devctl | 0x0003;
262 // 3:2 DACCTL[1:0] - DAC mode selection = 01: Immediate Update mode for DAC-configured ports.
263 devctl = devctl | 0x0004;
264 // 5:4 ADCCONV[1:0] - ADC conversion rate selection = 11: ADC conversion rate of 400ksps
265 devctl = devctl | 0x0030;
266 // 6 DACREF - DAC voltage reference selection = 1: Internal reference voltage
267 devctl = devctl | 0x0040;
268 // 7 THSHDN - Thermal shutdown enable = 1: Thermal shutdown function enabled.
269 devctl = devctl | 0x0080;
270 // 10:8 TMPCTL[2:0] - Temperature monitor selection = 001: Internal temperature monitor enabled
271 devctl = devctl | 0x0100;
272 // 11 TMPPER - Temperature conversion time control = 0 (Default)
273 // 12 RS_CANCEL - Temperature sensor series resistor cancellation mode = 0 (Default)
274 // 13 LPEN - Power mode selection = 0 (Default)
275 // 14 BRST - Serial interface burst-mode selection = 1: Contextual address incrementing mode
276 devctl = devctl | 0x4000;
277 // 15 RESET - Soft reset control = 0 (Default)
278
279 // Write the device configuration
280 if(WriteRegister(MAX11300_DEVCTL, devctl) == Result::ERR)
281 {
282 return Result::ERR;
283 }
284 // Verify our configuration was written...
285 if(ReadRegister(MAX11300_DEVCTL) != devctl)
286 {
287 return Result::ERR;
288 }
289
290 // Add a delay as recommended in the datasheet.
291 DelayUs(200);
292
293 // Set all pins to the default high impedance state...
294 for(uint8_t i = 0; i <= Pin::PIN_19; i++)
295 {
296 PinConfig pin_cfg;
297 pin_cfg.Defaults();
298 pin_configurations_[i] = pin_cfg;
299 SetPinConfig(static_cast<Pin>(i));
300 }
301
302 return Result::OK;
303 }

503 Result Update()
504 {
505 // Check first if were ready to TX/RX
506 if(!transport_.Ready())
507 return Result::OK;
508
509 if(dac_pin_count_ > 0)
510 {
511 // This is a burst transaction utilizing the contextual addressing
512 // scheme of the MAX11300. See the datasheet @ pp. 30
513 //
514 // We've prefixed the dac_buffer_ to point at the first dac pin to be written to.
515 // Subsequent DAC pins are written in their absolute order (0-19) if configured as such.
516 // For example:
517 // [1st_dac_pin_addr],[1st_dac_pin_msb],[1st_dac_pin_lsb],[2nd_dac_pin_msb],[2nd_dac_pin_lsb]...
518 //
519 // The size of the transaction is determined by the number of configured dac pins,
520 // plus one byte for the initial pin address.
521 //
522 // The datasheet recommends waiting 80us between DAC updates, in practice the appears to be
523 // per configured DAC pin. Here we inform the transport to wait at least N uS before transmitting
524 // again.
525 size_t tx_size = (dac_pin_count_ * 2) + 1;
526 if(transport_.Transmit(dac_buffer_, tx_size, (dac_pin_count_ * 40))
527 != Transport::Result::OK)
528 {
529 return Result::ERR;
530 }
531 }
532
533 if(adc_pin_count_ > 0)
534 {
535 // Reading ADC pins is a burst transaction approximately the same as the DAC transaction
536 // as described above...
537 size_t size = (adc_pin_count_ * 2) + 1;
538
539 uint8_t tx_buff[MAX_TRANSPORT_BUFFER_LENGTH] = {};
540 tx_buff[0] = adc_buffer_[0];
541 if(transport_.TransmitAndReceive(tx_buff, adc_buffer_, size)
542 != Transport::Result::OK)
543 {
544 adc_buffer_[0] = tx_buff[0];
545 return Result::ERR;
546 }
547 adc_buffer_[0] = tx_buff[0];
548 }
549
550 if(gpo_pin_count_ > 0)
551 {
552 // Writing GPO pins is a single 5 byte transaction, with the first byte being the
553 // the GPO data register, and the subsequent 4 bytes containing the state of the
554 // GPO ports to be written.
555 if(transport_.Transmit(gpo_buffer_, sizeof(gpo_buffer_), 0)
556 != Transport::Result::OK)
557 {
558 return Result::ERR;
559 }
560 }
561
562 if(gpi_pin_count_ > 0)
563 {
564 // Reading GPI pins is a single, 5 byte, full-duplex transaction with the first
565 // and only TX byte being the GPI register.
566 uint8_t tx_buff[sizeof(gpi_buffer_)] = {};
567 tx_buff[0] = gpi_buffer_[0];
568 if(transport_.TransmitAndReceive(
569 tx_buff, gpi_buffer_, sizeof(gpi_buffer_))
570 != Transport::Result::OK)
571 {
572 gpi_buffer_[0] = tx_buff[0];
573 return Result::ERR;
574 }
575 gpi_buffer_[0] = tx_buff[0];
576 }
577
578 return Result::OK;
579 }

589 static uint16_t VoltsTo12BitUint(float volts, DacVoltageRange range)
590 {
591 float vmax = 0;
592 float vmin = 0;
593 float vscaler = 0;
594 switch(range)
595 {
596 case DacVoltageRange::NEGATIVE_10_TO_0:
597 vmin = -10;
598 vmax = 0;
599 vscaler = 4095.0f / (vmax - vmin);
600 break;
601 case DacVoltageRange::NEGATIVE_5_TO_5:
602 vmin = -5;
603 vmax = 5;
604 vscaler = 4095.0f / (vmax - vmin);
605 break;
606 case DacVoltageRange::ZERO_TO_10:
607 vmin = 0;
608 vmax = 10;
609 vscaler = 4095.0f / (vmax - vmin);
610 break;
611 }
612 // Clamp...
613 if(volts > vmax)
614 volts = vmax;
615
616 if(volts < vmin)
617 volts = vmin;
618
619 return static_cast<uint16_t>((volts - vmin) * vscaler);
620 }

631 static float TwelveBitUintToVolts(uint16_t value, AdcVoltageRange range)
632 {
633 float vmax = 0;
634 float vmin = 0;
635 float vscaler = 0;
636 switch(range)
637 {
638 case AdcVoltageRange::NEGATIVE_10_TO_0:
639 vmin = -10;
640 vmax = 0;
641 vscaler = (vmax - vmin) / 4095;
642 break;
643 case AdcVoltageRange::NEGATIVE_5_TO_5:
644 vmin = -5;
645 vmax = 5;
646 vscaler = (vmax - vmin) / 4095;
647 break;
648 case AdcVoltageRange::ZERO_TO_10:
649 vmin = 0;
650 vmax = 10;
651 vscaler = (vmax - vmin) / 4095;
652 break;
653 case AdcVoltageRange::ZERO_TO_2P5:
654 vmin = 0;
655 vmax = 2.5;
656 vscaler = (vmax - vmin) / 4095;
657 break;
658 }
659 // Clamp...
660 if(value > 4095)
661 value = 4095;
662
663 return (value * vscaler) + vmin;
664 }

723 Result SetPinConfig(Pin pin)
724 {
725 uint16_t pin_func_cfg = 0x0000;
726
727 if(pin_configurations_[pin].mode != PinMode::NONE)
728 {
729 // Set the pin to high impedance mode before changing (as per the datasheet).
730 WriteRegister(MAX11300_FUNC_BASE + pin, 0x0000);
731 // According to the datasheet, the amount of time necessary for the pin to
732 // switch to high impedance mode depends on the prior configuration.
733 // The worst case recommended wait time seems to be 1ms.
734 DelayUs(1000);
735 }
736
737 // Apply the pin configuration
738 pin_func_cfg = pin_func_cfg
739 | static_cast<uint16_t>(pin_configurations_[pin].mode);
740
741 if(pin_configurations_[pin].mode == PinMode::ANALOG_OUT)
742 {
743 pin_func_cfg
744 |= static_cast<uint16_t>(pin_configurations_[pin].range.dac);
745 }
746 else if(pin_configurations_[pin].mode == PinMode::ANALOG_IN)
747 {
748 // In ADC mode we'll average 128 samples per Update
749 pin_func_cfg
750 = pin_func_cfg | 0x00e0
751 | static_cast<uint16_t>(pin_configurations_[pin].range.adc);
752 }
753 else if(pin_configurations_[pin].mode == PinMode::GPI)
754 {
755 // The DAC data register for that port needs to be set to the value corresponding to the
756 // intended input threshold voltage. Any input voltage above that programmed threshold is
757 // reported as a logic one. The input voltage must be between 0V and 5V.
758 // It may take up to 1ms for the threshold voltage to be effective
759 WriteRegister((MAX11300_DACDAT_BASE + pin),
760 MAX11300Driver::VoltsTo12BitUint(
761 pin_configurations_[pin].threshold,
762 pin_configurations_[pin].range.dac));
763 }
764 else if(pin_configurations_[pin].mode == PinMode::GPO)
765 {
766 // The ports DAC data register needs to be set first. It may require up to 1ms for the
767 // port to be ready to produce the desired logic one level.
768 WriteRegister((MAX11300_DACDAT_BASE + pin),
769 MAX11300Driver::VoltsTo12BitUint(
770 pin_configurations_[pin].threshold,
771 pin_configurations_[pin].range.dac));
772 }
773
774 // Write the configuration now...
775 if(WriteRegister(MAX11300_FUNC_BASE + pin, pin_func_cfg) != Result::OK)
776 {
777 return Result::ERR;
778 }
779
780 // Wait for 1ms as per the datasheet
781 DelayUs(1000);
782
783 // Verify our configuration was written
784 if(ReadRegister(MAX11300_FUNC_BASE + pin) != pin_func_cfg)
785 {
786 return Result::ERR;
787 }
788
789 // Update and re-index the pin configuration now...
790 UpdatePinConfig();
791
792 return Result::OK;
793 }

798 Result UpdatePinConfig()
799 {
800 // Zero everything out...
801 std::memset(dac_buffer_, 0, sizeof(dac_buffer_));
802 std::memset(adc_buffer_, 0, sizeof(adc_buffer_));
803 std::memset(gpi_buffer_, 0, sizeof(gpi_buffer_));
804 std::memset(gpo_buffer_, 0, sizeof(gpo_buffer_));
805
806 dac_pin_count_ = 0;
807 adc_pin_count_ = 0;
808 gpi_pin_count_ = 0;
809 gpo_pin_count_ = 0;
810
811 for(uint8_t i = 0; i <= Pin::PIN_19; i++)
812 {
813 Pin pin = static_cast<Pin>(i);
814
815 // Always reset the value pointer first...
816 pin_configurations_[i].value = nullptr;
817
818 if(pin_configurations_[i].mode == PinMode::ANALOG_OUT)
819 {
820 dac_pin_count_++;
821 if(dac_pin_count_ == 1)
822 {
823 // If this is the first pin of this type, we need to set
824 // the initial address of the dac_buffer_ to point at this pin.
825 // The ordering of subsequent pins is known by the MAX11300.
826 dac_buffer_[0] = (MAX11300_DACDAT_BASE + pin) << 1;
827 }
828 // set the pin_config.value to a pointer at the appropriate
829 // index of the dac_buffer...
830 pin_configurations_[i].value = reinterpret_cast<uint16_t*>(
831 &dac_buffer_[(2 * dac_pin_count_) - 1]);
832 }
833 else if(pin_configurations_[i].mode == PinMode::ANALOG_IN)
834 {
835 adc_pin_count_++;
836 if(adc_pin_count_ == 1)
837 {
838 // If this is the first pin of this type, we need to set
839 // the initial address of the adc_buffer_ to point at this pin.
840 // The ordering of subsequent pins is known by the MAX11300.
841 adc_buffer_[0] = ((MAX11300_ADCDAT_BASE + pin) << 1) | 1;
842 }
843 // set the pin_config.value to a pointer at the appropriate
844 // index of the adc_buffer...
845 pin_configurations_[i].value = reinterpret_cast<uint16_t*>(
846 &adc_buffer_[(2 * adc_pin_count_) - 1]);
847 }
848 else if(pin_configurations_[i].mode == PinMode::GPI)
849 {
850 gpi_pin_count_++;
851 gpi_buffer_[0] = (MAX11300_GPIDAT << 1) | 1;
852 }
853 else if(pin_configurations_[i].mode == PinMode::GPO)
854 {
855 gpo_pin_count_++;
856 gpo_buffer_[0] = (MAX11300_GPODAT << 1);
857 }
858 }
859
860 return Result::OK;
861 }

892 Result ReadRegister(uint8_t address, uint16_t* values, size_t size)
893 {
894 size_t rx_length = (size * 2) + 1;
895 uint8_t rx_buff[MAX_TRANSPORT_BUFFER_LENGTH] = {};
896 uint8_t tx_buff[MAX_TRANSPORT_BUFFER_LENGTH] = {};
897 tx_buff[0] = (address << 1) | 1;
898
899 if(transport_.TransmitAndReceive(tx_buff, rx_buff, rx_length)
900 != Transport::Result::OK)
901 {
902 return Result::ERR;
903 }
904
905 size_t rx_idx = 1;
906 for(size_t i = 0; i < size; i++)
907 {
908 values[i] = static_cast<uint16_t>((rx_buff[rx_idx] << 8)
909 + rx_buff[rx_idx + 1]);
910 rx_idx = rx_idx + 2;
911 }
912 return Result::OK;
913 }
